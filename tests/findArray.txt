programa pruebaPrograma ;

variables:
    matrix[3][4]: flotante;
    i, j: entero;
    n: flotante;

void funcion findValue(valor: flotante)
{
    i = 0;
    mientras (i < 2)  hacer {
        j = 0;
        mientras (j < 3) hacer {
            si (matrix[i][j] == valor) entonces {
                escribe("aqui esta");
            }
            j = j+1;
        }   
        i = i+1;
    }
}

principal () {
    escribe("Arreglo");

    desde i = 0 hasta 2 hacer {
        desde j = 0 hasta 3 hacer {
            matrix[i][j] = 2.0;
            escribe(matrix[i][j]);
        }
    }

    escribe("Que numero quieres encontrar");
    lee(n);
    findValue(n);
}
